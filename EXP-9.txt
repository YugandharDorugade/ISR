# Example: Simple content-based recommender using cosine similarity

from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# Step 1: Dataset
movies = [
    "Inception sci-fi thriller dream",
    "Interstellar sci-fi space drama",
    "Avengers action sci-fi heroes",
    "La La Land romance drama music"
]

# Step 2: TF-IDF Vectorization
vectorizer = TfidfVectorizer()
tfidf_matrix = vectorizer.fit_transform(movies)

# Step 3: Compute cosine similarity
similarity = cosine_similarity(tfidf_matrix)

# Step 4: Print similar movies
movie_index = 0  # Inception
print("Movies similar to:", movies[movie_index])
for i in range(len(movies)):
    if i != movie_index:
        print(f"  {movies[i]} â†’ Similarity: {round(similarity[movie_index][i], 2)}")
